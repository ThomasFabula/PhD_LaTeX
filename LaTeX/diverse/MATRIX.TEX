\left ( \begin{array}{llllll}
         C_{11}  &  C_{12}   &   C_{12}  &    0   &    0   &   0 \\
                 &  C_{11}   &   C_{12}  &    0   &    0   &   0 \\
                 &           &   C_{11}  &    0   &    0   &   0 \\
                 &           &           & C_{44} &    0   &   0 \\
                 &           &           &        & C_{44} &   0 \\
 \scriptstyle symmetrisch   &           &           &
& & C_{44}\\ \end{array}
\right )
\end{displaymath}
